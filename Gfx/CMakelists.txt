include(${CMAKE_SOURCE_DIR}/cmake/target_force_include.cmake)

file(GLOB_RECURSE GFX_COMMON_SOURCES CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.c
)
list(FILTER GFX_COMMON_SOURCES EXCLUDE REGEX ".*/src/dx11/.*")

file(GLOB_RECURSE GFX_SHARED_HEADERS CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/include/gfx/*.h
    ${CMAKE_CURRENT_SOURCE_DIR}/include/gfx/*.hpp
)

file(GLOB_RECURSE GFX_COMMON_HEADERS CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.hpp
)
list(FILTER GFX_COMMON_HEADERS EXCLUDE REGEX ".*/dx11/.*")

file(GLOB_RECURSE SHADER_FILES CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/shaders/*.hlsl
)

if (MSVC)
    foreach(SHADER ${SHADER_FILES})
        get_filename_component(SHADER_NAME ${SHADER} NAME_WE)
        if(SHADER_NAME MATCHES "^VS_")
            set_source_files_properties(${SHADER} PROPERTIES
                VS_SHADER_TYPE "Vertex"
                VS_SHADER_MODEL "5.0"
                VS_SHADER_ENTRYPOINT "main"
                VS_SHADER_OBJECT_FILE_NAME "$(OutDir)shaders/%(Filename).cso"
                VS_SHADER_DISABLE_OPTIMIZATIONS $<$<CONFIG:Debug>:ON>
                VS_SHADER_ENABLE_DEBUG $<$<CONFIG:Debug>:ON>
            )
        elseif(SHADER_NAME MATCHES "^PS_")
            set_source_files_properties(${SHADER} PROPERTIES
                VS_SHADER_TYPE "Pixel"
                VS_SHADER_MODEL "5.0"
                VS_SHADER_ENTRYPOINT "main"
                VS_SHADER_OBJECT_FILE_NAME "$(OutDir)shaders/%(Filename).cso"
                VS_SHADER_DISABLE_OPTIMIZATIONS $<$<CONFIG:Debug>:ON>
                VS_SHADER_ENABLE_DEBUG $<$<CONFIG:Debug>:ON>
            )
        endif()
    endforeach()
endif()

find_package(glm CONFIG REQUIRED)

add_library(gfx_objects OBJECT
    ${GFX_COMMON_SOURCES}
    ${GFX_COMMON_HEADERS}
)

target_include_directories(gfx_objects
    PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_compile_definitions(gfx_objects PUBLIC 
    GFX_EXPORTS
)

target_precompile_headers(gfx_objects
    PUBLIC  ${CMAKE_SOURCE_DIR}/PCH/pch.h
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/pch.h
)

target_link_libraries(gfx_objects
    PUBLIC
        Core
        fg::FrameGraph
        nvrhi
        glm::glm
)

if (RENDERER_BACKEND STREQUAL "DX11")
    add_subdirectory(src/dx11)
    target_compile_definitions(gfx_objects PRIVATE RENDERER_BACKEND_DX11)
else()
    message(FATAL_ERROR "Unknown backend: ${RENDERER_BACKEND}")
endif()

add_library(gfx SHARED
    $<TARGET_OBJECTS:gfx_objects>
    ${SHADER_FILES}
    ${GFX_SHARED_HEADERS}
)

target_include_directories(gfx
    PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_link_libraries(gfx PRIVATE gfx_objects)

target_compile_definitions(gfx PUBLIC 
    GFX_EXPORTS
)

if (RENDERER_BACKEND STREQUAL "DX11")
    target_compile_definitions(gfx PRIVATE RENDERER_BACKEND_DX11)
endif()

if (BUILD_TESTS)
    find_package(Catch2 CONFIG REQUIRED)

    file(GLOB_RECURSE GFX_TESTS_SOURCE CONFIGURE_DEPENDS
        ${CMAKE_CURRENT_SOURCE_DIR}/tests/*.cpp
    )
    
    add_executable(gfx_tests
        ${GFX_TESTS_SOURCE}
        ${GFX_TESTS_SOURCE}
        $<TARGET_OBJECTS:gfx_objects>
    )
    
    target_include_directories(gfx_tests
        PUBLIC  ${CMAKE_CURRENT_SOURCE_DIR}/tests
        PRIVATE
            ${CMAKE_CURRENT_SOURCE_DIR}/include
            ${CMAKE_CURRENT_SOURCE_DIR}/src
    )

    target_force_include(gfx_tests
        PRIVATE "${CMAKE_SOURCE_DIR}/PCH/pch.h" "${CMAKE_CURRENT_SOURCE_DIR}/src/pch.h"
    )
    
    target_link_libraries(gfx_tests PRIVATE gfx_objects Catch2::Catch2 Catch2::Catch2WithMain)

    if (RENDERER_BACKEND STREQUAL "DX11")
        target_compile_definitions(gfx_tests PRIVATE RENDERER_BACKEND_DX11)
    endif()
endif()
