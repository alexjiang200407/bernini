if (RENDERER_BACKEND STREQUAL "DX12")

    find_program(DXC_EXECUTABLE
        NAMES dxc.exe
        DOC "DirectX Shader Compiler (DXC)"
    )

    if(NOT DXC_EXECUTABLE)
        message(FATAL_ERROR "DXC not found. Make sure dxc.exe is in your PATH.")
    else()
        message(STATUS "Found DXC at ${DXC_EXECUTABLE}")
    endif()

    set(SHADER_OUT_DIR "${CMAKE_CURRENT_BINARY_DIR}/shaders")
    file(MAKE_DIRECTORY "${SHADER_OUT_DIR}")

    set(DX12_SHADERS "")

    foreach(SHADER ${SHADER_FILES})
        get_filename_component(SHADER_NAME ${SHADER} NAME_WE)

        if(SHADER_NAME MATCHES "^VS_")
            set(SHADER_TYPE "vs")
        elseif(SHADER_NAME MATCHES "^PS_")
            set(SHADER_TYPE "ps")
        elseif(SHADER_NAME MATCHES "^CS_")
            set(SHADER_TYPE "cs")
        else()
            message(WARNING "Unknown shader type for ${SHADER_NAME}, skipping")
            continue()
        endif()

        set(OUT_CSO "${SHADER_OUT_DIR}/${SHADER_NAME}.cso")

        add_custom_command(
            OUTPUT "${OUT_CSO}"
            COMMAND ${CMAKE_COMMAND} -E make_directory "${SHADER_OUT_DIR}"
            COMMAND ${DXC_EXECUTABLE}
                    -T ${SHADER_TYPE}_6_0
                    -Fo "${OUT_CSO}"
                    -Zi
                    -E ${SHADER_NAME} 
                    -Qembed_debug
                    "${SHADER}"
            DEPENDS "${SHADER}"
            COMMENT "Compiling DX12 shader ${SHADER_NAME} -> DXIL"
            VERBATIM
        )

        list(APPEND DX12_SHADERS "${OUT_CSO}")
    endforeach()

    set_source_files_properties(${SHADER_FILES} PROPERTIES VS_TOOL_OVERRIDE "None")
    set(GFX_SHADER_DIR "$<TARGET_FILE_DIR:gfx>/shaders")
    add_custom_target(CopyDX12Shaders ALL
        DEPENDS ${DX12_SHADERS}
        COMMAND ${CMAKE_COMMAND} -E copy_directory
                "${SHADER_OUT_DIR}"
                "${GFX_SHADER_DIR}"
        COMMENT "Copying DX12 shaders to gfx.dll folder"
    )
endif()
