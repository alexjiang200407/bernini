file(GLOB_RECURSE RENDERER_COMMON_SOURCES CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/src/common/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/common/*.c
)

file(GLOB_RECURSE RENDERER_SHARED_HEADERS CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/include/Renderer/*.h
    ${CMAKE_CURRENT_SOURCE_DIR}/include/Renderer/*.hpp
)

file(GLOB_RECURSE RENDERER_COMMON_HEADERS CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/src/common/*.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/common/*.hpp
)

add_library(Renderer STATIC
    ${RENDERER_SHARED_SOURCES}
    ${RENDERER_SHARED_HEADERS}
    ${RENDERER_COMMON_HEADERS}
)

target_precompile_headers(Renderer
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src/common/pch.h
    PUBLIC
        ${CMAKE_SOURCE_DIR}/PCH/pch.h
)

target_link_libraries(
    Renderer
    PRIVATE
        Core
        fg::FrameGraph
)

target_include_directories(Renderer
    PUBLIC  ${CMAKE_CURRENT_SOURCE_DIR}/include
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src
)

if (RENDERER_BACKEND STREQUAL "DX11")
    add_subdirectory(src/dx11)
else()
    message(FATAL_ERROR "Unknown backend: ${RENDERER_BACKEND}")
endif()

target_sources(Renderer PRIVATE ${RENDERER_COMMON_SOURCES} ${RENDERER_COMMON_HEADERS} ${RENDERER_SHARED_HEADERS})
