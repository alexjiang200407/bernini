set(IDL_CPP_OUT_DIR        "${CMAKE_SOURCE_DIR}/libs/bgl/src/idl")
set(IDL_PUBLIC_CPP_OUT_DIR "${CMAKE_SOURCE_DIR}/libs/bgl/include/bgl")
set(IDL_SLANG_OUT_DIR      "${CMAKE_SOURCE_DIR}/libs/bgl/shaders/src/idl")
set(IDL_SRC_ROOT           "${CMAKE_CURRENT_SOURCE_DIR}")

file(GLOB_RECURSE IDL_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/*.slang")

# Every module that emits a C++ mirror must be named in exactly one of the two lists below --
# CMake declares the .h as an OUTPUT, so it cannot glob for it. `scripts/gen_idl.py` (`just idl`)
# parses these same two lists, so the two generation paths cannot drift and emit different trees.
#
# A module in NEITHER list gets its Slang copy only, and no C++ header. That is correct for the
# interface- and generic-only modules -- IMaterial, IPbrMaterial, IAlphaTestMaterial, and the
# `Entry<T>` / `Range<T>` / `RangeWithCount<T>` offset primitives. A generic has no concrete layout
# to mirror, so idlgen skips it (and says so); their C++ counterparts in src/idl are **hand-written**
# and carry no generated banner. Listing one here would declare a CMake OUTPUT nothing produces.

# Internal: namespace `bgl::idl`, emitted into src/idl.
set(IDL_CPP_SOURCES
    Meshlet.slang
    VertexLayout.slang
    Submesh.slang
    CullView.slang
    CullStats.slang
    InstanceVisibility.slang
    Mesh.slang
    DebugRecord.slang
    Constants.slang
    PbrMaterial.slang
    LoosePbrMaterial.slang
    ChannelSource.slang
    TextureHandle.slang
    ErrorCode.slang
    DispatchArgs.slang
)

# Public engine API: namespace `bgl`, emitted into include/bgl. These are named in bgl's public
# headers (MaterialHandle, PsoType is used by ForwardPass and callers), so they cannot live behind
# `bgl::idl`. There is deliberately no `bgl::idl` copy of either -- a second, differently-namespaced
# definition of the same enum is a duplicate waiting to be included by mistake.
set(IDL_PUBLIC_CPP_SOURCES
    MaterialType.slang
    PsoType.slang
)

set(IDL_GENERATED_OUTPUTS "")

foreach(src ${IDL_SOURCES})
    file(RELATIVE_PATH rel "${IDL_SRC_ROOT}" "${src}")
    get_filename_component(rel_dir "${rel}" DIRECTORY)
    get_filename_component(name "${src}" NAME_WE)
    get_filename_component(file "${src}" NAME)

    if(rel_dir)
        set(rel_stem "${rel_dir}/${name}")
    else()
        set(rel_stem "${name}")
    endif()

    set(slang_out "${IDL_SLANG_OUT_DIR}/${rel_stem}.slang")
    set(outputs   "${slang_out}")
    set(cpp_args  "")

    if(file IN_LIST IDL_PUBLIC_CPP_SOURCES)
        # Public engine header: namespace `bgl`, emitted into include/bgl.
        set(cpp_args --cpp-out-dir "${IDL_PUBLIC_CPP_OUT_DIR}" --namespace "bgl")
        list(APPEND outputs "${IDL_PUBLIC_CPP_OUT_DIR}/${rel_stem}.h")
    elseif(file IN_LIST IDL_CPP_SOURCES)
        set(cpp_args --cpp-out-dir "${IDL_CPP_OUT_DIR}")
        list(APPEND outputs "${IDL_CPP_OUT_DIR}/${rel_stem}.h")
    endif()

    add_custom_command(
        OUTPUT  ${outputs}
        COMMAND bgl_idlgen
                --src-root "${IDL_SRC_ROOT}"
                --slang-out-dir "${IDL_SLANG_OUT_DIR}"
                ${cpp_args}
                -I "${IDL_SRC_ROOT}"
                "${src}"
        DEPENDS bgl_idlgen "${src}"
        COMMENT "idlgen ${rel}"
        VERBATIM
    )

    list(APPEND IDL_GENERATED_OUTPUTS ${outputs})
endforeach()

add_custom_target(bgl_idl_generate ALL DEPENDS ${IDL_GENERATED_OUTPUTS})
